================================================================================
                        EXAMPLE CYPHER QUERIES
================================================================================

-- Get full brand context (for content generation)
MATCH (b:Brand {id: $brand_id})
OPTIONAL MATCH (b)-[:HAS_LOGO]->(logo:Logo)
OPTIONAL MATCH (b)-[cr:HAS_COLOR]->(c:ColorNode)
OPTIONAL MATCH (b)-[sr:HAS_STYLE]->(s:StyleNode)
OPTIONAL MATCH (b)-[:HAS_COMPOSITION]->(comp:CompositionNode)
OPTIONAL MATCH (b)-[:SELLS]->(p:ProductNode)
OPTIONAL MATCH (b)-[:HAS_CHARACTER]->(ch:CharacterNode)
OPTIONAL MATCH (b)-[:LEARNED]->(lp:LearnedPreference)
RETURN b, logo, collect(DISTINCT c) as colors, 
       collect(DISTINCT s) as styles, comp, 
       collect(DISTINCT p) as products,
       collect(DISTINCT ch) as characters,
       collect(DISTINCT lp) as learned_preferences

-- Create learned preference from feedback
MATCH (b:Brand {id: $brand_id})
MATCH (g:Generation {id: $generation_id})
CREATE (lp:LearnedPreference {
    id: $pref_id,
    attribute: $attribute,
    preference_type: $type,
    value: $value,
    reasoning: $reasoning,
    confidence: $confidence,
    created_at: datetime()
})
CREATE (b)-[:LEARNED {confidence: $confidence}]->(lp)
CREATE (lp)-[:DERIVED_FROM]->(g)

-- Get brand's learned preferences
MATCH (b:Brand {id: $brand_id})-[:LEARNED]->(lp:LearnedPreference)
WHERE lp.confidence >= 0.5
RETURN lp ORDER BY lp.confidence DESC

-- Get generation history with feedback
MATCH (b:Brand {id: $brand_id})-[:GENERATED]->(g:Generation)
OPTIONAL MATCH (g)-[:RECEIVED_FEEDBACK]->(f:Feedback)
RETURN g, collect(f) as feedback
ORDER BY g.created_at DESC
LIMIT 10
